212       *                                               FUNCIONAMIENTO DE LA CALCULADORA
213       * -----------------------------------------------------------------------------------------------------------------------------
214       */
215      public void actionPerformed(ActionEvent evento_pulsar) {    // Metodo con el que manejamos los eventos cuando pulsamos los botones de la calculadora
216  
217          /**
218           * -----------------------------------------------PULSAR BOTON DE NUMERO------------------------------------------
219           */
220  
221          if (evento_pulsar.getSource() == boton_0) { // Verifica si pulsamos el boton0, se ejecutara el siguiente contenido
222  
223              if (pantalla_resultado.getText().equals("0")){  //Comprueba si pantalla_resultado esta a 0, esto esta para que no ponga 0 si ya hay un 0, asi evita cosas como "0000"
224                  cadenaNumeros = "0";    // Se establece en cadenaNumeros el numero 0 en pantalla
225              }
226              else {
227                  cadenaNumeros +="0";    // Si pantalla_resultado no esta en 0, se agrega un 0 al final de la cadena almacenada en cadena mueros
228              }
229              pantalla_resultado.setText(cadenaNumeros);  // Se actualiza la pantalla_resultado con el contenido de cadenaNumeros
230              activado=true;  // Se activa la entrada de numeros, aunque por defecto ya venia en true
231              punto = true;   //Permite agregar el punto
232          }
233  
234          /**
235           * Hacemos los mismos pasos con los demas botones
236           */
237  
238          else if (evento_pulsar.getSource() == boton_1) { if (pantalla_resultado.getText().equals("1")){cadenaNumeros = "1"; }
239          else {cadenaNumeros +="1";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
240          else if (evento_pulsar.getSource() == boton_2) { if (pantalla_resultado.getText().equals("2")){cadenaNumeros = "2"; }
241          else {cadenaNumeros +="2";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
242          else if (evento_pulsar.getSource() == boton_3) { if (pantalla_resultado.getText().equals("3")){cadenaNumeros = "3"; }
243          else {cadenaNumeros +="3";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
244          else if (evento_pulsar.getSource() == boton_4) { if (pantalla_resultado.getText().equals("4")){cadenaNumeros = "4"; }
245          else {cadenaNumeros +="4";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
246          else if (evento_pulsar.getSource() == boton_5) { if (pantalla_resultado.getText().equals("5")){cadenaNumeros = "5"; }
247          else {cadenaNumeros +="5";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
248          else if (evento_pulsar.getSource() == boton_6) { if (pantalla_resultado.getText().equals("6")){cadenaNumeros = "6"; }
249          else {cadenaNumeros +="6";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
250          else if (evento_pulsar.getSource() == boton_7) { if (pantalla_resultado.getText().equals("7")){cadenaNumeros = "7"; }
251          else {cadenaNumeros +="7";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
252          else if (evento_pulsar.getSource() == boton_8) { if (pantalla_resultado.getText().equals("8")){cadenaNumeros = "8"; }
253          else {cadenaNumeros +="8";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
254          else if (evento_pulsar.getSource() == boton_9) { if (pantalla_resultado.getText().equals("9")){cadenaNumeros = "9"; }
255          else {cadenaNumeros +="9";} pantalla_resultado.setText(cadenaNumeros);activado=true;}
256  
257          /**
258           * -----------------------------------------------FUNCIONES DE CADA OPERACION BASICA------------------------------------------
259           */
260  
261          if (evento_pulsar.getSource() == boton_sumar) {     // Comprueba si el evento viene del boton sumar
262  
263              if (activado == true){  // Comprueba que activado este en true
264                  primerNumero = Double.parseDouble(cadenaNumeros);   // Guarda en primerNumero el contenido de cadenaNumeros y lo copnvierte en un Double.
265                  pantalla_operaciones.setText(cadenaNumeros+ " + ");     // Muestra en pantalla_operaciones la opearcion que se va realizar
266                  cadenaNumeros = "";     //Limpia la variable cadenaNumeros para despues ingresar otro numero nuevo
267                  operacion = "sumar";    // Establece "sumar" en operacion, asi luego indicara que esta operacion es una suma
268                  activado = false;   // Indica que la calculadora no esta lista para realizar operacion, todavia ya que falta el segundo numero
269              }
270          }
271  
272          /**
273           * Hacemos los mismos pasos con las demas operaciones, vemos que es basicamente igual pero cambia el nombre y el "+" por "-"
274           */
275  
276          //OPERACION RESTAR
277  
278          if (evento_pulsar.getSource() == boton_restar) {
279              if (activado == true){
280                  primerNumero = Double.parseDouble(cadenaNumeros);
281                  pantalla_operaciones.setText(cadenaNumeros+ " - ");
282                  cadenaNumeros = "";
283                  operacion = "restar";
284  
285                  activado = false;
286              }
287          }
288  
289          //OPERACION MULITPLICAR
290  
291          if (evento_pulsar.getSource() == boton_multiplicar) {
292              if (activado == true){
293                  primerNumero = Double.parseDouble(cadenaNumeros);
294                  pantalla_operaciones.setText(cadenaNumeros+ " x ");
295                  cadenaNumeros = "";
296                  operacion = "multiplicar";
297  
298                  activado = false;
299              }
300          }
301  
302          //OPERACION DIVIDIR
303  
304          if (evento_pulsar.getSource() == boton_dividir) {
305              if (activado == true){
306                  primerNumero = Double.parseDouble(cadenaNumeros);
307                  pantalla_operaciones.setText(cadenaNumeros+ " / ");
308                  cadenaNumeros = "";
309                  operacion = "dividir";
310  
311                  activado = false;
312              }
313          }
314  
315          /**
316           * -----------------------------------------------FUNCIONES PARA PUNTO Y BORRAR-----------------------------------------
317           */
318  
319          /**
320           * AGREGAR DECIMAL
321           */
322  
323          if (evento_pulsar.getSource() == boton_punto) {     //Verifica que el evento viene del boton punto
324  
325              if (punto == true){ // Verifica que punto esta en true
326                  if (cadenaNumeros.equals("")){  // Comprueba si la cadena de numeros esta vacia
327                      cadenaNumeros = "0.";   // Si esta vacia, agrega un "." despues del 0. En lugar de poner solo el punto sin un numero ni nada
328                  }
329                  else {
330                      cadenaNumeros += ".";   // Si no esta vacia, añade el punto al contenido que tenia puesto
331                  }
332              }
333              pantalla_resultado.setText(cadenaNumeros);  // Actualiza la pantalla_resultado para mostrar el numero con el decimal agregado
334              punto = false;  // Se pone en false, indica que no se puede ingresar otro punto decimal a la varibale actual, ya que ya tiene un decimal puesto
335          }
336  
337          /**
338           * BORRAR TODO
339           */
340  
341          if (evento_pulsar.getSource() == boton_c) {     // Verficia que el boton pulsado es el C
342  
343              pantalla_operaciones.setText("");   // Limpia la pantalla de operaciones
344              pantalla_resultado.setText(""); // Limpia la pantalla de resultado
345              operacion = "nula"; // Indica que no se esta realizando ninguna opearacion
346              cadenaNumeros = ""; // Limpia la cadena de numeros
347              activado = true;    // Indica que la calculadora esta lista para una nueva opearcion
348              punto = true;   // Permite agregar un punto decimal otra vez
349  
350          }
351  
352          if (evento_pulsar.getSource() == boton_ce) {    // Verifica que el boton pulsado es el CE
353  
354              int tamaño = cadenaNumeros.length();    // Calcula la longitud de la cadena de numeros y lo guarda en la variable tamaño
355              if (tamaño > 0){    // Veficia si es mayor a 0, si lo es procedera a hacer la tarea siguiente
356  
357                  if (tamaño == 1){   // Si el tamaño es igual a 1, la cadena de numeros automaticamente se volvera a 0
358                      cadenaNumeros = "0";
359                  }
360                  else {
361                      cadenaNumeros = cadenaNumeros.substring(0,cadenaNumeros.length()-1);    // Si es menor a 1, se eliminara el ultimo caracterde la cadena de nuemros. Esto con la ayuda de substring
362                  }
363                  pantalla_resultado.setText(cadenaNumeros);  //La pantalla resultado mostrara el valor actual de la cadena de numeros
364              }
365          }
366  
367          /**
368           * -----------------------------------------------FUNCIONES DE CADA OPERACION COMPLEJA-----------------------------------------
369           */
370  
371          /**
372           * OPERACION RAIZ
373           */
374  
375          if (evento_pulsar.getSource() == boton_raiz) {  // Verifica que el boton pulsado ese el de raiz cuadrada
376  
377              primerNumero = Double.parseDouble(cadenaNumeros);   // Guarda en primerNumero el contenido de la cadena de numero convertida en un Double
378              pantalla_operaciones.setText("sqrt("+cadenaNumeros+")");    // Muestra en pantalla de opearciones la operacion
379              resultado = Math.sqrt(primerNumero);    // Con la funcion Math.sqrt se calcula la raiz cuadrada del primer numero, y lo guarda en resultado
380              pantalla_resultado.setText(String.valueOf(resultado));  // Se actualiza para mostrar el resultado de la raiz cuadrada
381              cadenaNumeros = String.valueOf(resultado);  // Actualiza con el resultado de esa raiz cuadrada para que asi pueda hacer operaciones adicionales con ese resuiltado
382              punto = true;   // Permite agregar decimal si el usuario lo desea
383  
384          }
385  
386          /**
387           * OPERACION POTENCIA ELEVADA A 2
388           * es basicamente lo mismo que el de la raiz cuadrada, pero cambiamos la operacion del resultado
389           */
390  
391          if (evento_pulsar.getSource() == boton_potencia) {
392              primerNumero = Double.parseDouble(cadenaNumeros);
393              pantalla_operaciones.setText("sqr("+cadenaNumeros+")");
394              resultado = Math.pow(primerNumero,2);
395              pantalla_resultado.setText(String.valueOf(resultado));
396              cadenaNumeros = String.valueOf(resultado);
397              punto = true;
398          }
399  
400          /**
401           * -----------------------------------------------CALCULAR RESULTADOS OPERACIONES-----------------------------------------
402           */
403  
404          /**
405           * BOTON IGUAL
406           */
407          if (evento_pulsar.getSource() == boton_igual) { // Verificamos que el boton pulsado es el de igual
408              double segundoNumero;   // Creamos una nueva variable que guardara el segundo numero introducido por el usuario
409              if (operacion.equals("nula")){  // Verifica si operacione esta igual a nula
410                  pantalla_resultado.setText(cadenaNumeros);  // Muestra el contenido de cadena numeros en la pantalla de resultado
411              }
412  
413                  /**
414                   * CALCULAR SUMA
415                   */
416                  else if (operacion.equals("sumar")) {   // Verifica que la opearcion esta indicada como "sumar", esto indica que el usuario ha seleccionado antes la operacion de sumar
417
